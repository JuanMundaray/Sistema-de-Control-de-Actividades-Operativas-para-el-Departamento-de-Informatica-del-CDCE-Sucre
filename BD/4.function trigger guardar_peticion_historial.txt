CREATE FUNCTION actividades.guardar_peticion_historial() RETURNS trigger LANGUAGE 'plpgsql' COST 100 VOLATILE NOT LEAKPROOF AS $BODY$ begin insert into actividades.historial_peticiones values(new.id_peticion,new.nombre_peticion, new.departamento_peticion, new.detalles_peticion, new.fecha_peticion);return new;END $BODY$;
