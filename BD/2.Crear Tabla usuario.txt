create table actividades.usuario(id_usuario serial, nombre_usuario varchar(100), nombre varchar(100), cedula varchar(15),contrasena varchar(100),tipo_usuario varchar(100),fecha_creacion date, id_departamento integer,primary key(id_usuario),constraint "nombreusuario_UNI" unique(nombre_usuario),constraint "cedula_UNI" unique(cedula),FOREIGN KEY(id_departamento) REFERENCES actividades.departamentos(id_departamento));
