CREATE TABLE IF NOT EXISTS actividades.peticiones(id_peticion serial NOT NULL,nombre_peticion character varying(100),departamento_peticion integer,detalles_peticion character varying(512),fecha_peticion date,id_usuario integer,PRIMARY KEY (id_peticion), FOREIGN KEY(id_usuario) REFERENCES actividades.usuario(id_usuario) ON DELETE CASCADE, FOREIGN KEY(departamento_peticion) REFERENCES actividades.departamentos(id_departamento));
